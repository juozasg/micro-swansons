    Overview

    A delivered message from a subscription.

    nc = NATS::Connection.new("demo.nats.io")
nc.subscribe("foo") do |msg|
  puts "Received '#{msg}'"
  puts "Subject is #{msg.subject}"
  puts "Reply Subject is #{msg.reply}"
  puts "Raw Data is #{msg.data}"
end
